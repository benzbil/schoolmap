คู่มือการ Deploy ระบบ

วิธีการนำระบบไปใช้งานจริงบน GitHub Pages

ขั้นตอนการเตรียมไฟล์

ขั้นตอนที่ 1: เปลี่ยนชื่อไฟล์

เมื่อดาวน์โหลดไฟล์แล้ว ให้เปลี่ยนชื่อดังนี้:

school_map_system_v6.html → index.html
school_map_system_admin_v6.html → admin.html

สำคัญ: ต้องเปลี่ยนชื่อไฟล์ให้ถูกต้องเพื่อให้ GitHub Pages ทำงานได้

ขั้นตอนที่ 2: สร้าง Repository บน GitHub

  • เข้าสู่ GitHub และสร้าง Repository ใหม่
  • ตั้งชื่อ Repository เช่น school-map-system
  • เลือก Public Repository
  • อัปโหลดไฟล์ index.html และ admin.html

ขั้นตอนที่ 3: เปิดใช้งาน GitHub Pages

  1. ไปที่ Settings ของ Repository
  2. เลื่อนลงไปหา "Pages" ในเมนูซ้าย
  3. ในส่วน Source เลือก "Deploy from a branch"
  4. เลือก Branch: main และ Folder: / (root)
  5. คลิก Save

ขั้นตอนที่ 4: แก้ไข URL ในโค้ด

หลังจาก Deploy แล้ว ต้องแก้ไข URL ในไฟล์ทั้งสองเพื่อให้เชื่อมต่อกันได้:

ในไฟล์ index.html หาบรรทัด:

function openAdminInterface() {
    window.open('school_map_system_admin_v6.html', '_blank');
}

แก้เป็น:

function openAdminInterface() {
    window.open('https://[USERNAME].github.io/[REPO]/admin.html', '_blank');
}

ในไฟล์ admin.html หาบรรทัด:

function openUserInterface() {
    window.open('school_map_system_v6.html', '_blank');
}

แก้เป็น:

function openUserInterface() {
    window.open('https://[USERNAME].github.io/[REPO]/index.html', '_blank');
}

แทนที่ [USERNAME] ด้วยชื่อผู้ใช้ GitHub และ [REPO] ด้วยชื่อ Repository

ขั้นตอนที่ 5: ทดสอบระบบ

  • เข้าใช้งานผ่าน URL: https://[USERNAME].github.io/[REPO]/
  • ทดสอบการค้นหาห้องเรียน
  • ทดสอบการตั้งจุดเริ่มต้นและเส้นทาง
  • ทดสอบการเข้าสู่ Admin Panel (admin/1234)
  • ทดสอบการเพิ่ม/แก้ไข/ลบข้อมูล

ข้อกำหนดและข้อแนะนำ

📋 ข้อกำหนดการใช้งาน

  • เบราว์เซอร์ที่รองรับ: Chrome, Firefox, Safari, Edge (เวอร์ชันล่าสุด)
  • ใช้งานได้ทั้งบนคอมพิวเตอร์และมือถือ
  • ไม่ต้องติดตั้ง Server หรือ Database
  • ข้อมูลจัดเก็บใน localStorage ของเบราว์เซอร์

💡 ข้อแนะนำเพิ่มเติม

  • สำรองข้อมูลเป็นประจำผ่านฟีเจอร์ Export ในระบบ
  • ใช้ Google Sheets สำหรับ Live Mode เพื่อการทำงานร่วมกัน
  • ปรับแต่งข้อมูลห้องเรียนให้ตรงกับโรงเรียนจริง
  • ทดสอบระบบอย่างสม่ำเสมอหลังการปรับปรุง

function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { // ✅ เพิ่ม CORS headers const output = ContentService.createTextOutput(); output.setMimeType(ContentService.MimeType.JSON); try { // ✅ Enhanced parameter extraction let params = e.parameter || {}; // ✅ ถ้าไม่มี parameters จาก e.parameter ลองดู postData if (Object.keys(params).length === 0 && e.postData) { console.log('No parameters in e.parameter, checking postData...'); console.log('PostData type:', e.postData.type); console.log('PostData contents:', e.postData.contents); if (e.postData.type === 'application/x-www-form-urlencoded') { // Parse URL-encoded data manually const postDataString = e.postData.contents; const postParams = {}; postDataString.split('&').forEach(pair => { const [key, value] = pair.split('='); if (key && value) { postParams[decodeURIComponent(key)] = decodeURIComponent(value); } }); params = postParams; console.log('Parsed postData params:', params); } } console.log('=== REQUEST DEBUG ==='); console.log('Method:', e.parameter ? 'GET/Parameter' : 'POST/PostData'); console.log('Final parameters:', params); console.log('Parameter keys:', Object.keys(params)); const action = params.action; const sheetName = params.sheet || 'Buildings'; } // ✅ Handle different actions switch(action.toLowerCase()) { case 'getdata': return getData(sheetName, output); case 'adddata': return addData(sheetName, params, output); case 'updatedata': return updateData(sheetName, params, output); case 'deletedata': return deleteData(sheetName, params.id, output); default: return output.setContent(JSON.stringify({ success: false, error: `Invalid action: "${action}"`, valid_actions: ['getData', 'addData', 'updateData', 'deleteData'], received_action: action, all_parameters: params })); const realphoto = e.parameter.realphoto || ''; // หาแถวที่มี id = 'maps' หรือสร้างใหม่ const data = mapSheet.getDataRange().getValues(); let found = false; for (let i = 1; i < data.length; i++) { if (data[i][0] === 'maps') { mapSheet.getRange(i + 1, 2, 1, 2).setValues([[floorplan, realphoto]]); found = true; break; } } if (!found) { mapSheet.appendRow(['maps', floorplan, realphoto]); } return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'Map images updated successfully' })).setMimeType(ContentService.MimeType.JSON); } function getMapImages() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const mapSheet = ss.getSheetByName('MapImages'); if (!mapSheet) { return ContentService.createTextOutput(JSON.stringify({ success: true, data: { floorplan: '', realphoto: '' } })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ success: true, data: { floorplan: '', realphoto: '' } })).setMimeType(ContentService.MimeType.JSON); } function getData(sheetName, output) { try { console.log('getData called with sheetName:', sheetName); const spreadsheet = SpreadsheetApp.getActiveSpreadsheet(); console.log('Spreadsheet name:', spreadsheet.getName()); const sheet = spreadsheet.getSheetByName(sheetName); if (!sheet) { const availableSheets = spreadsheet.getSheets().map(s => s.getName()); throw new Error(`Sheet "${sheetName}" not found. const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) { throw new Error(`Sheet "${sheetName}" not found`); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const codeIndex = headers.indexOf('code'); if (codeIndex === -1) { throw new Error('Column "code" not found in sheet headers: ' + headers.join(', ')); } if (!params.code) { throw new Error('Missing "code" parameter for update operation'); } let updated = false; let updatedRow = -1; for (let i = 1; i < data.length; i++) { if (data[i][codeIndex] === params.code) { headers.forEach((header, index) => { if (params[header] !== undefined) { sheet.getRange(i + 1, index + 1).setValue(params[header]); } }); updated = true; updatedRow = i + 1; break; } } return output.setContent(JSON.stringify({ success: true, message: updated ? 'Data updated successfully' : 'No matching record found', updated: updated, updated_row: updatedRow, search_code: params.code, timestamp: new Date().toISOString() })); } catch (error) { console.error('updateData error:', error); return output.setContent(JSON.stringify({ success: false, error: error.toString(), context: 'updateData function' })); } } function deleteData(sheetName, id, output) { try { console.log('deleteData called with:', sheetName, id); if (!id) { throw new Error('Missing "id" parameter for delete operation'); } const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) { throw new Error(`Sheet "${sheetName}" not found`); } const data = sheet.getDataRange().getValues(); const headers = data[0]; const codeIndex = headers.indexOf('code'); if (codeIndex === -1) { throw new Error('Column "code" not found in sheet headers: ' + headers.join(', ')); } let deleted = false; let deletedRow = -1; for (let i = 1; i < data.length; i++) { if (data[i][codeIndex] === id) { sheet.deleteRow(i + 1); deleted = true; deletedRow = i + 1; break; } } return output.setContent(JSON.stringify({ success: true, message: deleted ? 'Data deleted successfully' : 'No matching record found', deleted: deleted, deleted_row: deletedRow, search_id: id, timestamp: new Date().toISOString() })); } catch (error) { console.error('deleteData error:', error); return output.setContent(JSON.stringify({ success: false, error: error.toString(), context: 'deleteData function' })); } }
